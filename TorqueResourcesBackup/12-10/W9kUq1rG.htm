  148. function getMouseAdjustAmount(%val)
  149. {
  150.    // based on a default camera FOV of 90'
  151.    return(%val * ($cameraFov / 90) * 0.01) * $pref::Input::LinkMouseSensitivity;
  152. }

  154. function getGamepadAdjustAmount(%val)
  155. {
  156.    // based on a default camera FOV of 90'
  157.    return(%val * ($cameraFov / 90) * 0.01) * 10.0;
  158. }

  160. function yaw(%val)
  161. {
  162.    %yawAdj = getMouseAdjustAmount(%val);
  163.    if(ServerConnection.isControlObjectRotDampedCamera())
  164.    {
  165.       // Clamp and scale
  166.       %yawAdj = mClamp(%yawAdj, -m2Pi()+0.01, m2Pi()-0.01);

  168.    }
  169.

  219. function gamepadYaw(%val)
  220. {
  221.    %yawAdj = getGamepadAdjustAmount(%val);
  222.    if(ServerConnection.isControlObjectRotDampedCamera())
  223.    {
  224.       // Clamp and scale
  225.       %yawAdj = mClamp(%yawAdj, -m2Pi()+0.01, m2Pi()-0.01);
  226.       %yawAdj *= 0.5;
  227.    }
  228.  
  229.    if(%yawAdj > 0)                                             
  230.    {
  231.       $mvYawLeftSpeed = %yawAdj;
  232.       $mvYawRightSpeed = 0;
  233.    }
  234.    else                                                                
  235.    {
  236.       $mvYawLeftSpeed = 0;
  237.       $mvYawRightSpeed = -%yawAdj;
  238.    }
  239. }

  241. function gamepadPitch(%val)
  242. {
  243.    %pitchAdj = getGamepadAdjustAmount(%val);
  244.    if(ServerConnection.isControlObjectRotDampedCamera())
  245.    {
  246.       // Clamp and scale
  247.       %pitchAdj = mClamp(%pitchAdj, -m2Pi()+0.01, m2Pi()-0.01);

  256.    else                                                        
  257.    {
  258.       $mvPitchDownSpeed = 0;
  259.       $mvPitchUpSpeed = %pitchAdj;              //- and + swap for INVERTED look                BrittEdit
  260.    }
  261. }

  336. function gamepadFire(%val)
  337. {
  338.    if(%val > 0.1 && !$gamepadFireTriggered)
  339.    {
  340.       $gamepadFireTriggered = true;
  341.       $mvTriggerCount0++;
  342.           rumble(gamepad, 0.25, 0.25);//deviceName, Left Motor(low-frequency),Right Motor(high-frequency)//Rumble Start BrittEdit
  343.    }
  344.    else if(%val <= 0.1 && $gamepadFireTriggered)
  345.    {
  346.       $gamepadFireTriggered = false;
  347.       $mvTriggerCount0++;
  348.           rumble(gamepad, 0, 0);                                                        //Rumble Stop BrittEdit

  370. // Zoom and FOV functions
  371. //------------------------------------------------------------------------------
  372.  
  373. if($Player::CurrentFOV $= "")
  374.    $Player::CurrentFOV = $pref::Player::DefaultFOV / 2;
  375.  
  376. // toggleZoomFOV() works by dividing the CurrentFOV by 2.  Each time that this
  377. // toggle is hit it simply divides the CurrentFOV by 2 once again.  If the
  378. // FOV is reduced below a certain threshold then it resets to equal half of the
  379. // DefaultFOV value.  This gives us 4 zoom levels to cycle through.
  380.  
  381. function toggleZoomFOV()
  382. {
  383.     $Player::CurrentFOV = $Player::CurrentFOV / 2;
  384.  
  385.     if($Player::CurrentFOV < 5)
  386.         resetCurrentFOV();
  387.  
  388.     if(ServerConnection.zoomed)
  389.       setFOV($Player::CurrentFOV);
  390.     else
  391.     {
  392.       setFov(ServerConnection.getControlCameraDefaultFov());
  393.     }
  394. }
  395.  
  396. function resetCurrentFOV()
  397. {
  398.    $Player::CurrentFOV = ServerConnection.getControlCameraDefaultFov() / 2;
  399. }
  400.  
  401. function turnOffZoom()
  402. {
  403.    ServerConnection.zoomed = false;
  404.    setFov(ServerConnection.getControlCameraDefaultFov());
  405.    Reticle.setVisible(true);
  406.    zoomReticle.setVisible(false);
  407.  
  408.    // Rather than just disable the DOF effect, we want to set it to the level's
  409.    // preset values.
  410.    //DOFPostEffect.disable();
  411.    ppOptionsUpdateDOFSettings();
  412. }
  413.  
  414. function setZoomFOV(%val)
  415. {
  416.    if(%val)
  417.       toggleZoomFOV();
  418. }
  419.  
  420. function toggleZoom(%val)
  421. {
  422.    if (%val)
  423.    {
  424.       ServerConnection.zoomed = true;
  425.       setFov($Player::CurrentFOV);
  426.       Reticle.setVisible(false);
  427.       zoomReticle.setVisible(true);
  428.  
  429.       DOFPostEffect.setAutoFocus( true );
  430.       DOFPostEffect.setFocusParams( 0.5, 0.5, 50, 500, -5, 5 );
  431.       DOFPostEffect.enable();
  432.    }
  433.    else
  434.    {
  435.       turnOffZoom();
  436.    }
  437. }

  741. // Trace a line along the direction the crosshair is pointing
  742. // If you find a car with a player in it...eject them
  743. function carjack()
  744. {
  745.    %player = LocalClientConnection.getControlObject();
  746.  
  747.    if (%player.getClassName() $= "Player")
  748.    {
  749.       %eyeVec = %player.getEyeVector();
  750.  
  751.       %startPos = %player.getEyePoint();
  752.       %endPos = VectorAdd(%startPos, VectorScale(%eyeVec, 1000));
  753.  
  754.       %target = ContainerRayCast(%startPos, %endPos, $TypeMasks::VehicleObjectType);
  755.  
  756.       if (%target)
  757.       {
  758.          // See if anyone is mounted in the car's driver seat
  759.          %mount = %target.getMountNodeObject(0);
  760.  
  761.          // Can only carjack bots
  762.          // remove '&& %mount.getClassName() $= "AIPlayer"' to allow you
  763.          // to carjack anyone/anything
  764.          if (%mount && %mount.getClassName() $= "AIPlayer")
  765.          {
  766.             commandToServer('carUnmountObj', %mount);
  767.          }
  768.       }
  769.    }
  770. }
